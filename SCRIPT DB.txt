CREATE TABLE TICKET (
 CD_TICKET_ID INT AUTO_INCREMENT PRIMARY KEY
,DS_TIPO VARCHAR(20) NOT NULL

,CD_MATRICULA INT NOT NULL
,DS_AREA VARCHAR(50) NOT NULL
,DS_POSTO VARCHAR(50)
,DS_ORIGEM VARCHAR(10)
,DS_CLASSIFICACAO VARCHAR(20)
,DS_PROBLEMA VARCHAR(2000) NOT NULL
,DS_ACAO VARCHAR(2000)
,DS_SOLUCAO VARCHAR(2000)
,NM_RESPONSAVEL VARCHAR(100) NOT NULL
,DS_STATUS VARCHAR(10) DEFAULT 'EM ANALISE'
,DT_EMISSAO DATETIME
,NM_USUARIO VARCHAR(100)
